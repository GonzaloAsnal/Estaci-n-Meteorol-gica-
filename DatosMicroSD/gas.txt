    function actualizarGrafico() {
            var hs = document.getElementById("hs").value;
            var horas, valores;

            // Obtener los datos según la opción seleccionada
            if (hs === "1h"){
                horas = !!HORA;
                valores = !!VALORESH;
            } else if (hs === "1d") {
                horas = !!HORAS;
                valores = !!VALORESHS;
            } else if (hs === "7d") {
                horas = !!7DIAS;
                valores = !!VALORES7D;
            } else if (hs === "1m") {
                horas = !!1MES;
                valores = !!VALORES1M;
            } else if (hs === "1a") {
                horas = !!12MES;
                valores = !!VALORES12M;
            }
            // Llamar a la función graficar con los datos obtenidos
            graficar(horas, valores);
            console.log(hs)
        }

    function graficar(horas, valores) {
            var ctx = document.getElementById("myChart").getContext('2d');
            if (window.myChart != undefined) {
                window.myChart.destroy();
            }
            window.myChart = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: horas,
                    datasets: [{
                        label: 'Valores',
                        data: valores,
                        backgroundColor: 'rgba(255, 99, 132, 0.2)',
                        borderColor: 'rgba(255, 99, 132, 1)',
                        borderWidth: 3,
	                    tension: 0.3
                    }]
                },
                options: {
                    scales: {
                        yAxes: [{
                            ticks: {
                                beginAtZero: true
                            }
                        }]
                    }
                }
            });
        }

        // Llamar a actualizarGrafico al cargar la página para tener el gráfico inicial
        actualizarGrafico();
